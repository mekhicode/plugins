#!/bin/bash
IN_DIR=~/download/test
OUT_DIR=~/download/crop
EXT=".jpg"
mkdir $OUT_DIR
cp $IN_DIR/* $OUT_DIR
cd $OUT_DIR
for file in `ls $OUT_DIR/*$EXT`; do
  ls -al $file
  convert $file -gravity center -crop 640x340+0+0 -resize 640x340^ $file
done
